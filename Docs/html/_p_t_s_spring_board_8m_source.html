81 -(void)layoutSubviews {
82  /*Layouts the item container...*/
83  itemsContainer.frame = CGRectMake(0.0f, 0.0f, self.frame.size.width, self.frame.size.height);
84 
85  /*...the page-control.*/
86  pageControl.frame = CGRectMake(0.0f, self.frame.size.height-20.0f, self.frame.size.width, 20.0f);
87 
88  /*Now layouts the content-size of the item-container...*/
89  NSInteger numberOfItems = [[self delegate] numberOfItemsInSpringboard:self];
90 
91  CGSize sizePerItem = CGSizeMake(100.0f, 100.f) ;
92 
93  NSInteger numberOfRows = floor(self.frame.size.width / sizePerItem.width);
94  NSInteger numberOfColumns = floor(self.frame.size.height / sizePerItem.height);
95  NSInteger numberOfItemsPerPage = numberOfRows * numberOfColumns;
96  NSInteger numberOfPages = ceil((float)numberOfItems/numberOfItemsPerPage);
97 
98 
99  if (pageControl.numberOfPages > numberOfPages) {
100  pageControl.currentPage = pageControl.currentPage - 1;
101  }
102 
103  pageControl.numberOfPages = numberOfPages;
104 
105  [itemsContainer setContentSize:CGSizeMake(numberOfPages*itemsContainer.frame.size.width, itemsContainer.frame.size.height)];
106 
107  CGFloat rowGap = ([[self superview] frame].size.width - numberOfRows * sizePerItem.width)/numberOfRows;
108  CGFloat columnGap = ([[self superview] frame].size.height - numberOfColumns * sizePerItem.height)/numberOfColumns;
109 
110  /*...and the positions of the individual items.*/
111  for (PTSSpringBoardItem * item in itemsInOrder) {
112  NSInteger index = [itemsInOrder indexOfObject:item];
113 
114  if (index != NSNotFound) {
115  NSInteger pageNumber = floor((float)(index/numberOfItemsPerPage));
116 
117  NSInteger collumnIdx = floor((float) (index % numberOfItemsPerPage) / numberOfRows);
118  NSInteger rowIdx = (index % numberOfRows);
119 
120  /*If animation is in editing mode (except for the first layouting), repositioning of all items is animated.*/
121  if ([self isEditing]) {
122  [UIView animateWithDuration:0.25f animations:^(void){
123  [item setFrame:CGRectMake((rowIdx * (100.0f + rowGap)) + (pageNumber*itemsContainer.frame.size.width), (collumnIdx * (100.0f + columnGap)), sizePerItem.width, sizePerItem.height)];
124  }];
125  } else {
126  [item setFrame:CGRectMake((rowIdx * (100.0f + rowGap)) + (pageNumber*itemsContainer.frame.size.width), (collumnIdx * (100.0f + columnGap)), sizePerItem.width, sizePerItem.height)];
127  }
128  }
129  }
130 
131  [super layoutSubviews];
132 }

142 -(BOOL)item:(PTSSpringBoardItem*)item collidesWithOtherItem:(PTSSpringBoardItem **)collisionItem {
143  for (PTSSpringBoardItem * spItem in itemsInOrder) {
144  if (spItem != item && [item isMovable] && [spItem isMovable]) {
145  CGPoint p1 = item.center;
146  CGPoint p2 = spItem.center;
147 
148  if (sqrt((p1.x - p2.x) * (p1.x - p2.x) + (p1.y - p2.y) * (p1.y - p2.y)) <= 25.0f) {
149  *collisionItem = spItem;
150  return YES;
151  }
152  }
153  }
154 
155  return NO;
156 }

179 -(void)updateItemAtIndex:(NSInteger)index {
180  PTSSpringBoardItem * item = [[self dataSource] springboard:self itemForIndex:index];
181 
182  if ([itemsInOrder count] >= (index + 1) && item) {
183  PTSSpringBoardItem * oldItem = [itemsInOrder objectAtIndex:index];

185  [oldItem removeFromSuperview];
186  [itemsInOrder removeObject:oldItem];

192  [itemsContainer addSubview:item];

265 -(void)springboardIsDraging:(PTSSpringBoardItem *)item {
266  /*A springboard item reports that a touchEvent is moving, which will cause the corresponding item to move. This method takes care of two things:
267 
268  a) It checks for collisions with other items in the view. If one is detected, and the delegate responds to -(void)springboard:movedItemAtIndex:toIndex:, it will cause the two items to be exchanged internally.
269  b) It checks, wheather the item is move across the borders to another page.
270 
271  The internal exchange of items upon collision is only temporary for display reasons. The data source will be informed of the exchange with the -(void)springboard: movedItem: atIndex: toIndex: method. It has to take care of the reordering, otherwise the items position will be reset, when dragging ends.
272  */
273  PTSSpringBoardItem * collisionItem;
274 
275  /*Checks for collision.*/
276  if ([self item:item collidesWithOtherItem:&collisionItem] && [[self delegate] respondsToSelector:@selector(springboard:movedItem:atIndex:toIndex:)]) {
277 
278  oldIndex = [itemsInOrder indexOfObject:item];
279  newIndex = [itemsInOrder indexOfObject:collisionItem];
280 
281  if (oldIndex != NSNotFound && newIndex != NSNotFound) {
282  [itemsInOrder exchangeObjectAtIndex:newIndex withObjectAtIndex:oldIndex];
283  [[self delegate] springboard:self movedItem:item atIndex:oldIndex toIndex:newIndex];
284 
285  [self setNeedsLayout];
286  }
287  }
288 
289  /*Checks for moving an item across a border.*/
290  CGRect contentFrame = CGRectMake(0.0f, 0.0f, itemsContainer.contentSize.width, itemsContainer.contentSize.height);
291  if (!CGRectContainsRect(itemsContainer.frame, item.frame) && CGRectContainsRect(contentFrame, item.frame)) {
292 
293  CGRect visibleFrame = CGRectMake(itemsContainer.contentOffset.x, itemsContainer.contentOffset.y, itemsContainer.contentSize.width, itemsContainer.contentSize.height);
294 
295  CGRect nextPageRect;
296 
297  if (item.frame.origin.x > visibleFrame.origin.x) {
298  nextPageRect = CGRectMake(((pageControl.currentPage + 1) * itemsContainer.bounds.size.width), itemsContainer.bounds.origin.y, itemsContainer.bounds.size.width, itemsContainer.bounds.size.height);
299  } else {
300  nextPageRect = CGRectMake(((pageControl.currentPage - 1) * itemsContainer.bounds.size.width), itemsContainer.bounds.origin.y, itemsContainer.bounds.size.width, itemsContainer.bounds.size.height);
301  }
302 
303  [itemsContainer scrollRectToVisible:nextPageRect animated:YES];
304  }
